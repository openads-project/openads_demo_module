37template <typename T>
38void OpenadsDemoModule::declareAndLoadParameter(const std::string& name,
39 T& param,
40 const std::string& description,
41 const bool add_to_auto_reconfigurable_params,
42 const bool is_required,
43 const bool read_only,
44 const std::optional<double>& from_value,
45 const std::optional<double>& to_value,
46 const std::optional<double>& step_value,
47 const std::string& additional_constraints) {
48 rcl_interfaces::msg::ParameterDescriptor param_desc;
49 param_desc.description = description;
50 param_desc.additional_constraints = additional_constraints;
51 param_desc.read_only = read_only;
52
53 auto type = rclcpp::ParameterValue(param).get_type();
54
55 if (from_value.has_value() && to_value.has_value()) {
56 if constexpr (std::is_integral_v<T>) {
57 rcl_interfaces::msg::IntegerRange range;
58 range.set__from_value(static_cast<T>(from_value.value())).set__to_value(static_cast<T>(to_value.value()));
59 if (step_value.has_value()) range.set__step(static_cast<T>(step_value.value()));
60 param_desc.integer_range = {range};
61 } else if constexpr (std::is_floating_point_v<T>) {
62 rcl_interfaces::msg::FloatingPointRange range;
63 range.set__from_value(static_cast<T>(from_value.value())).set__to_value(static_cast<T>(to_value.value()));
64 if (step_value.has_value()) range.set__step(static_cast<T>(step_value.value()));
65 param_desc.floating_point_range = {range};
66 } else {
67 RCLCPP_WARN(this->get_logger(), "Parameter type of parameter '%s' does not support specifying a range", name.c_str());
68 }
69 }
70
71 this->declare_parameter(name, type, param_desc);
72
73 try {
74 param = this->get_parameter(name).get_value<T>();
75 std::stringstream ss;
76 ss << "Loaded parameter '" << name << "': ";
77 if constexpr (is_vector_v<T>) {
78 ss << "[";
79 for (const auto& element : param) ss << element << (&element != &param.back() ? ", " : "");
80 ss << "]";
81 } else {
82 ss << param;
83 }
84 RCLCPP_INFO_STREAM(this->get_logger(), ss.str());
85 } catch (rclcpp::exceptions::ParameterUninitializedException&) {
86 if (is_required) {
87 RCLCPP_FATAL_STREAM(this->get_logger(), "Missing required parameter '" << name << "', exiting");
88 exit(EXIT_FAILURE);
89 } else {
90 std::stringstream ss;
91 ss << "Missing parameter '" << name << "', using default value: ";
92 if constexpr (is_vector_v<T>) {
93 ss << "[";
94 for (const auto& element : param) ss << element << (&element != &param.back() ? ", " : "");
95 ss << "]";
96 } else {
97 ss << param;
98 }
99 RCLCPP_WARN_STREAM(this->get_logger(), ss.str());
100 this->set_parameters({rclcpp::Parameter(name, rclcpp::ParameterValue(param))});
101 }
102 }
103
104 if (add_to_auto_reconfigurable_params) {
105 std::function<void(const rclcpp::Parameter&)> setter = [&param](const rclcpp::Parameter& p) { param = p.get_value<T>(); };
106 auto_reconfigurable_params_.push_back(std::make_tuple(name, setter));
107 }
108}

110rcl_interfaces::msg::SetParametersResult OpenadsDemoModule::parametersCallback(const std::vector<rclcpp::Parameter>& parameters) {
111 for (const auto& param : parameters) {
112 for (auto& auto_reconfigurable_param : auto_reconfigurable_params_) {
113 if (param.get_name() == std::get<0>(auto_reconfigurable_param)) {
114 std::get<1>(auto_reconfigurable_param)(param);
115 RCLCPP_INFO(this->get_logger(), "Reconfigured parameter '%s' to: %s", param.get_name().c_str(),
116 param.value_to_string().c_str());
117 break;
118 }
119 }
120 }
121
122 rcl_interfaces::msg::SetParametersResult result;
123 result.successful = true;
124
125 return result;
126}

128void OpenadsDemoModule::setup() {
129 // callback for dynamic parameter configuration
130 parameters_callback_ =
131 this->add_on_set_parameters_callback(std::bind(&OpenadsDemoModule::parametersCallback, this, std::placeholders::_1));
132
133 // subscriber for handling incoming messages
134 subscriber_ = this->create_subscription<geometry_msgs::msg::PointStamped>(
135 "~/input", 10, std::bind(&OpenadsDemoModule::topicCallback, this, std::placeholders::_1));
136 RCLCPP_INFO(this->get_logger(), "Subscribed to '%s'", subscriber_->get_topic_name());
137
138 // publisher for publishing outgoing messages
139 publisher_ = this->create_publisher<geometry_msgs::msg::PointStamped>("~/output", 10);
140 RCLCPP_INFO(this->get_logger(), "Publishing to '%s'", publisher_->get_topic_name());
141
142 // service server for handling service calls
143 service_server_ = this->create_service<std_srvs::srv::SetBool>(
144 "~/service", std::bind(&OpenadsDemoModule::serviceCallback, this, std::placeholders::_1, std::placeholders::_2));
145
146 // action server for handling action goal requests
147 action_server_ = rclcpp_action::create_server<openads_demo_module_interfaces::action::Fibonacci>(
148 this, "~/action", std::bind(&OpenadsDemoModule::actionHandleGoal, this, std::placeholders::_1, std::placeholders::_2),
149 std::bind(&OpenadsDemoModule::actionHandleCancel, this, std::placeholders::_1),
150 std::bind(&OpenadsDemoModule::actionHandleAccepted, this, std::placeholders::_1));
151
152 // timer for repeatedly invoking a callback
153 timer_ = this->create_wall_timer(std::chrono::duration<double>(1.0), std::bind(&OpenadsDemoModule::timerCallback, this));
154
155 // setup diagnostic updater
156 diagnostic_updater_.setHardwareID("none");
157 diagnostic_updater_.add("Health", this, &OpenadsDemoModule::health);
158
159 // add diagnostic task for monitoring topic subscription with a moving average over min. 5 incoming messages based on expected minimum frequency
160 const int topic_diagnostic_frequency_window_size =
161 std::ceil(5 / (diagnostic_updater_.getPeriod().seconds() * topic_diagnostic_config_.min_frequency));
162 topic_diagnostic_ = std::make_unique<diagnostic_updater::TopicDiagnostic>(
163 "~/input", diagnostic_updater_,
164 diagnostic_updater::FrequencyStatusParam(&topic_diagnostic_config_.min_frequency, &topic_diagnostic_config_.max_frequency,
165 0.0, topic_diagnostic_frequency_window_size),
166 diagnostic_updater::TimeStampStatusParam(topic_diagnostic_config_.min_acceptable_timestamp_delta,
167 topic_diagnostic_config_.max_acceptable_timestamp_delta));
168
169 // add diagnostic task for monitoring topic publisher with a moving average over min. 5 incoming messages based on expected minimum frequency
170 const int diagnosed_publisher_frequency_window_size =
171 std::ceil(5 / (diagnostic_updater_.getPeriod().seconds() * diagnosed_publisher_config_.min_frequency));
172 diagnosed_publisher_ = std::make_unique<diagnostic_updater::DiagnosedPublisher<geometry_msgs::msg::PointStamped>>(
173 publisher_, diagnostic_updater_,
174 diagnostic_updater::FrequencyStatusParam(&diagnosed_publisher_config_.min_frequency,
175 &diagnosed_publisher_config_.max_frequency, 0.0,
176 diagnosed_publisher_frequency_window_size),
177 diagnostic_updater::TimeStampStatusParam(diagnosed_publisher_config_.min_acceptable_timestamp_delta,
178 diagnosed_publisher_config_.max_acceptable_timestamp_delta));
179}

226void OpenadsDemoModule::actionExecute(
227 const std::shared_ptr<rclcpp_action::ServerGoalHandle<openads_demo_module_interfaces::action::Fibonacci>> goal_handle) {
228 RCLCPP_INFO(this->get_logger(), "Executing action goal");
229
230 // define a sleeping rate between computing individual Fibonacci numbers
231 rclcpp::Rate loop_rate(1);
232
233 // create handy accessors for the action goal, feedback, and result
234 const auto goal = goal_handle->get_goal();
235 auto feedback = std::make_shared<openads_demo_module_interfaces::action::Fibonacci::Feedback>();
236 auto result = std::make_shared<openads_demo_module_interfaces::action::Fibonacci::Result>();
237
238 // initialize the Fibonacci sequence
239 auto& partial_sequence = feedback->partial_sequence;
240 partial_sequence.push_back(0);
241 partial_sequence.push_back(1);
242
243 // compute the Fibonacci sequence up to the requested order n
244 for (int i = 1; i < goal->order && rclcpp::ok(); ++i) {
245 // cancel, if requested
246 if (goal_handle->is_canceling()) {
247 result->sequence = feedback->partial_sequence;
248 goal_handle->canceled(result);
249 RCLCPP_INFO(this->get_logger(), "Action goal canceled");
250 return;
251 }
252
253 // compute the next Fibonacci number
254 partial_sequence.push_back(partial_sequence[i] + partial_sequence[i - 1]);
255
256 // publish the current sequence as action feedback
257 goal_handle->publish_feedback(feedback);
258 RCLCPP_INFO(this->get_logger(), "Publishing action feedback");
259
260 // sleep before computing the next Fibonacci number
261 loop_rate.sleep();
262 }
263
264 // finish by publishing the action result
265 if (rclcpp::ok()) {
266 result->sequence = partial_sequence;
267 goal_handle->succeed(result);
268 RCLCPP_INFO(this->get_logger(), "Goal succeeded");
269 }
270}

272void OpenadsDemoModule::timerCallback() {
273 RCLCPP_INFO(this->get_logger(), "Timer triggered");

275 // TODO(unknown): Remove this demonstration of health status and implement real health checks using `setHealth()`
276 if (health_.status == diagnostic_msgs::msg::DiagnosticStatus::ERROR) {

279 {{"uptime", std::to_string(this->get_clock()->now().seconds())}});
280 health_.status = diagnostic_msgs::msg::DiagnosticStatus::WARN;
281 } else if (health_.status == diagnostic_msgs::msg::DiagnosticStatus::WARN) {

286 setHealth(diagnostic_msgs::msg::DiagnosticStatus::OK,

288 health_.status = diagnostic_msgs::msg::DiagnosticStatus::STALE;
289 } else {
290 setHealth(diagnostic_msgs::msg::DiagnosticStatus::STALE, "Node performance level cannot be assessed");
291 health_.status = diagnostic_msgs::msg::DiagnosticStatus::ERROR;
292 }
