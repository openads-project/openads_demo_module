39void Ros2DemoNode::declareAndLoadParameter(const std::string& name, T& param, const std::string& description,
40 const bool add_to_auto_reconfigurable_params, const bool is_required,
41 const bool read_only, const std::optional<double>& from_value,
42 const std::optional<double>& to_value,
43 const std::optional<double>& step_value,
44 const std::string& additional_constraints) {
45 rcl_interfaces::msg::ParameterDescriptor param_desc;
46 param_desc.description = description;
47 param_desc.additional_constraints = additional_constraints;
48 param_desc.read_only = read_only;
49
50 auto type = rclcpp::ParameterValue(param).get_type();
51
52 if (from_value.has_value() && to_value.has_value()) {
53 if constexpr (std::is_integral_v<T>) {
54 rcl_interfaces::msg::IntegerRange range;
55 range.set__from_value(static_cast<T>(from_value.value())).set__to_value(static_cast<T>(to_value.value()));
56 if (step_value.has_value()) range.set__step(static_cast<T>(step_value.value()));
57 param_desc.integer_range = {range};
58 } else if constexpr (std::is_floating_point_v<T>) {
59 rcl_interfaces::msg::FloatingPointRange range;
60 range.set__from_value(static_cast<T>(from_value.value())).set__to_value(static_cast<T>(to_value.value()));
61 if (step_value.has_value()) range.set__step(static_cast<T>(step_value.value()));
62 param_desc.floating_point_range = {range};
63 } else {
64 RCLCPP_WARN(this->get_logger(), "Parameter type of parameter '%s' does not support specifying a range",
65 name.c_str());
66 }
67 }
68
69 this->declare_parameter(name, type, param_desc);
70
71 try {
72 param = this->get_parameter(name).get_value<T>();
73 std::stringstream ss;
74 ss << "Loaded parameter '" << name << "': ";
75 if constexpr (is_vector_v<T>) {
76 ss << "[";
77 for (const auto& element : param) ss << element << (&element != &param.back() ? ", " : "");
78 ss << "]";
79 } else {
80 ss << param;
81 }
82 RCLCPP_INFO_STREAM(this->get_logger(), ss.str());
83 } catch (rclcpp::exceptions::ParameterUninitializedException&) {
84 if (is_required) {
85 RCLCPP_FATAL_STREAM(this->get_logger(), "Missing required parameter '" << name << "', exiting");
86 exit(EXIT_FAILURE);
87 } else {
88 std::stringstream ss;
89 ss << "Missing parameter '" << name << "', using default value: ";
90 if constexpr (is_vector_v<T>) {
91 ss << "[";
92 for (const auto& element : param) ss << element << (&element != &param.back() ? ", " : "");
93 ss << "]";
94 } else {
95 ss << param;
96 }
97 RCLCPP_WARN_STREAM(this->get_logger(), ss.str());
98 this->set_parameters({rclcpp::Parameter(name, rclcpp::ParameterValue(param))});
99 }
100 }
101
102 if (add_to_auto_reconfigurable_params) {
103 std::function<void(const rclcpp::Parameter&)> setter = [&param](const rclcpp::Parameter& p) {
104 param = p.get_value<T>();

106 auto_reconfigurable_params_.push_back(std::make_tuple(name, setter));

230void Ros2DemoNode::actionExecute(
231 const std::shared_ptr<rclcpp_action::ServerGoalHandle<ros2_demo_package_interfaces::action::Fibonacci>>
232 goal_handle) {
233 RCLCPP_INFO(this->get_logger(), "Executing action goal");
234
235 // define a sleeping rate between computing individual Fibonacci numbers
236 rclcpp::Rate loop_rate(1);
237
238 // create handy accessors for the action goal, feedback, and result
239 const auto goal = goal_handle->get_goal();
240 auto feedback = std::make_shared<ros2_demo_package_interfaces::action::Fibonacci::Feedback>();
241 auto result = std::make_shared<ros2_demo_package_interfaces::action::Fibonacci::Result>();
242
243 // initialize the Fibonacci sequence
244 auto& partial_sequence = feedback->partial_sequence;
245 partial_sequence.push_back(0);
246 partial_sequence.push_back(1);
247
248 // compute the Fibonacci sequence up to the requested order n
249 for (int i = 1; i < goal->order && rclcpp::ok(); ++i) {
250 // cancel, if requested
251 if (goal_handle->is_canceling()) {
252 result->sequence = feedback->partial_sequence;
253 goal_handle->canceled(result);
254 RCLCPP_INFO(this->get_logger(), "Action goal canceled");
255 return;
256 }
257
258 // compute the next Fibonacci number
259 partial_sequence.push_back(partial_sequence[i] + partial_sequence[i - 1]);
260
261 // publish the current sequence as action feedback
262 goal_handle->publish_feedback(feedback);
263 RCLCPP_INFO(this->get_logger(), "Publishing action feedback");
264
265 // sleep before computing the next Fibonacci number
266 loop_rate.sleep();
267 }
268
269 // finish by publishing the action result
270 if (rclcpp::ok()) {
271 result->sequence = partial_sequence;
272 goal_handle->succeed(result);
273 RCLCPP_INFO(this->get_logger(), "Goal succeeded");
274 }
275}
